3 分钟快速开始

1

安装

npm install -g @toobit_ai/toobit-trade-mcp @toobit_ai/toobit-trade-cli
2

配置

toobit config init

# 或手动创建 ~/.toobit/config.toml
default_profile = "live"

[profiles.live]
api_key = "your-api-key"
secret_key = "your-secret-key"
3

交易

# 连接 AI 客户端
toobit-trade-mcp setup --client cursor

# 或直接在终端使用
toobit market ticker --symbol BTCUSDT
toobit spot place --symbol BTCUSDT \
  --side BUY --type MARKET --quantity 0.001
